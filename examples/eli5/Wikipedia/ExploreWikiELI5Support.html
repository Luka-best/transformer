def make_support(question, es_index_name='english_wiki40b_passages_100w'):
    q = question.lower()
    banned = ['how', 'why', 'what', 'where', 'which', 'do', 'does', 'is', '?', 'eli5', 'eli5:']
    q = ' '.join([w for w in q.split() if w not in banned])
    response = es_client.search(
        index = es_index_name,
        body = {
            "query": {
                "multi_match": {
                    "query": q,
                    "fields": [
                        "title",
                        "section",
                        "content^2",
                    ],
                    "type": "cross_fields",
                }
            }
        }
    )
    hits = response['hits']['hits']
    doc = '<P> ' + ' <P> '.join([hit['_source']['content'] for hit in hits])
    sources = [(hit['_id'], hit['_score']) for hit in hits]
    return (doc, sources)
